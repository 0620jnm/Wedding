    <script>
        // --- 設定區 ---
        const GAS_URL = "你的GAS部署網址貼在這裡"; // <--- 填入第一步產生的網址

        const fileInput = document.getElementById('file-upload');
        const statusMsg = document.getElementById('status-msg');
        const previewContainer = document.getElementById('preview-container');

        fileInput.addEventListener('change', async function(e) {
            const files = e.target.files;
            if (files.length === 0) return;

            statusMsg.innerText = "準備上傳中...";
            
            for (let i = 0; i < files.length; i++) {
                const file = files[i];
                
                // 1. 前端即時預覽
                const reader = new FileReader();
                reader.onload = function(event) {
                    const div = document.createElement('div');
                    div.className = 'photo-placeholder';
                    div.style.backgroundImage = `url(${event.target.result})`;
                    div.innerHTML = '';
                    previewContainer.insertBefore(div, previewContainer.firstChild);
                    if (previewContainer.children.length > 6) previewContainer.lastElementChild.remove();
                };
                reader.readAsDataURL(file);

                // 2. 轉換為 Base64 並上傳至 GAS
                try {
                    statusMsg.innerText = `正在上傳第 ${i+1}/${files.length} 張...`;
                    const base64Data = await toBase64(file);
                    const payload = {
                        filename: file.name,
                        contentType: file.type,
                        base64: base64Data.split(',')[1]
                    };

                    const response = await fetch(GAS_URL, {
                        method: 'POST',
                        body: JSON.stringify(payload)
                    });
                    
                    const result = await response.json();
                    if (result.result === "success") {
                        statusMsg.innerText = "上傳成功！感謝分享";
                    } else {
                        throw new Error(result.error);
                    }
                } catch (err) {
                    console.error(err);
                    statusMsg.innerText = "上傳失敗，請稍後再試";
                }
            }
        });

        const toBase64 = file => new Promise((resolve, reject) => {
            const reader = new FileReader();
            reader.readAsDataURL(file);
            reader.onload = () => resolve(reader.result);
            reader.onerror = error => reject(error);
        });
    </script>
